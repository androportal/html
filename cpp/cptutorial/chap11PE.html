              , size_t len1 )
    : CString(cs, len1), s_original(s1), pos( pos1 ) {}

    //   CSubstring copy constructor
    CSubstring( const CSubstring& s1) : CString( s1 ),
         s_original( s1.s_original) , pos( s1.pos ) {}
};
